struct Point {
    x: i32,
    y: i32
}

func point(x: i32, y: i32): Point {
    return Point { x: x, y: y };
}

func fib(n: i32): i32 {
    if n < 2 {
        return 1;
    } else {
        return fib(n - 1) + fib(n - 2);
    }
}

func sum(s: *i32, n: i32) {
    *s = *s + n;
}

func main() {
    var p = point(9, 7);
    print_point(&p);
    print_int(p.x);
    print_int(p.y);
    print_int(p.x + p.y);
}
